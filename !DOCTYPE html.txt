<script>
    pdfjsLib.GlobalWorkerOptions.workerSrc = 'https://cdnjs.cloudflare.com/ajax/libs/pdf.js/3.4.120/pdf.worker.min.js';

    let pdfImageSrc = ""; 
    let cropper;
    let currentCropTarget = ""; 
    let qrCodeInstance = null;

    function updateCard() {
        document.getElementById('cardName').innerText = document.getElementById('inName').value;
        document.getElementById('cardDob').innerText = document.getElementById('inDob').value;
        document.getElementById('cardGender').innerText = document.getElementById('inGender').value;
        document.getElementById('cardMobile').innerText = document.getElementById('inMobile').value;
        document.getElementById('cardAadhar').innerText = document.getElementById('inAadhar').value;
        document.getElementById('cardFarmerId').innerText = document.getElementById('inFarmerId').value;
        document.getElementById('cardAddress').innerText = document.getElementById('inAddress').value;
        
        generateAutoQR();
    }

    function generateAutoQR() {
        const qrContainer = document.getElementById("cardQr");
        qrContainer.innerHTML = ""; 
        
        const qrData = `Name: ${document.getElementById('inName').value}\nFarmerID: ${document.getElementById('inFarmerId').value}\nMobile: ${document.getElementById('inMobile').value}`;
        
        new QRCode(qrContainer, {
            text: qrData,
            width: 75,
            height: 75,
            correctLevel: QRCode.CorrectLevel.H
        });
    }

    function processPDF(event) {
        const file = event.target.files[0];
        if (!file || file.type !== "application/pdf") return alert("कृपया केवल सही PDF फाइल ही चुनें!");

        const fileReader = new FileReader();
        fileReader.onload = function() {
            const typedarray = new Uint8Array(this.result);
            pdfjsLib.getDocument(typedarray).promise.then(function(pdf) {
                pdf.getPage(1).then(function(page) {
                    const viewport = page.getViewport({ scale: 2.0 }); 
                    const canvas = document.createElement('canvas');
                    const ctx = canvas.getContext('2d');
                    canvas.height = viewport.height;
                    canvas.width = viewport.width;

                    const renderContext = { canvasContext: ctx, viewport: viewport };
                    page.render(renderContext).promise.then(function() {
                        pdfImageSrc = canvas.toDataURL('image/jpeg');
                        alert("PDF सफलतापूर्वक लोड हो गया है! अब आप फोटो या खेत का विवरण क्रॉप कर सकते हैं।");
                    });
                });
            });
        };
        fileReader.readAsArrayBuffer(file);
    }

    function openCropModal(target) {
        if (!pdfImageSrc) return alert("कृपया पहले एक PDF फाइल अपलोड करें!");
        currentCropTarget = target;
        
        const modal = document.getElementById('cropModal');
        const image = document.getElementById('imageToCrop');
        image.src = pdfImageSrc;
        
        modal.style.display = 'flex';

        if (cropper) cropper.destroy();

        const aspectRatio = (target === 'photo') ? 105 / 130 : NaN; 

        cropper = new Cropper(image, {
            aspectRatio: aspectRatio,
            viewMode: 1,
            autoCropArea: 0.8
        });
    }

    function closeModal() {
        document.getElementById('cropModal').style.display = 'none';
        if (cropper) cropper.destroy();
    }

    function saveCrop() {
        if (!cropper) return;
        
        const canvas = cropper.getCroppedCanvas();
        const croppedImgSrc = canvas.toDataURL('image/jpeg');

        if (currentCropTarget === 'photo') {
            document.getElementById('cardPhoto').src = croppedImgSrc;
        } else if (currentCropTarget === 'land') {
            document.getElementById('cardLandDetails').src = croppedImgSrc;
        }

        closeModal();
    }

    function downloadBothSides() {
        const farmerName = document.getElementById('inName').value || 'Farmer';
        
        html2canvas(document.getElementById('frontCard'), { scale: 3, useCORS: true }).then(canvas => {
            let link = document.createElement('a');
            link.download = farmerName + '_Front_Card.jpg';
            link.href = canvas.toDataURL('image/jpeg', 1.0);
            link.click();
        });

        setTimeout(() => {
            html2canvas(document.getElementById('backCard'), { scale: 3, useCORS: true }).then(canvas => {
                let link = document.createElement('a');
                link.download = farmerName + '_Back_Card.jpg';
                link.href = canvas.toDataURL('image/jpeg', 1.0);
                link.click();
            });
        }, 600);
    }

    window.onload = function() {
        generateAutoQR();
    };
</script>
